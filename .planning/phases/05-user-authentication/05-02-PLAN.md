    ```kotlin
    class LoginHandler(
        private val userRepository: UserRepository,
        private val sessionRegistry: SessionRegistry  // Review 修复: 移除 SnowflakeIdGenerator（不用于创建用户）
    ) : Handler<LoginReq, LoginResp> {

        override val method: String = "user/login"

        override suspend fun handle(req: LoginReq): LoginResp {
            // 场景 1: Token 重连（AUTH-02）
            if (req.hasToken()) {
                val token = req.token
                val existingSession = sessionRegistry.validate(token)
                if (existingSession != null) {
                    // Review 修复：复用现有 Token，不生成新 Token
                    // 避免 Session 孤儿化——旧 Session 在 Redis 中存活，新 Token 被生成，旧 Token 泄露
                    val user = userRepository.findById(existingSession.userId)
                        ?: throw UserException(BizCode.USER_NOT_FOUND)
                    return buildLoginResp(existingSession.userId, existingSession.token)
                }
            }

            // 场景 2: 用户名+密码登录
            val username = req.username ?: throw UserException(BizCode.INVALID_PARAM)
            val password = req.password ?: throw UserException(BizCode.INVALID_PARAM)
            val user = userRepository.findByUsername(username)
                ?: throw UserException(BizCode.USER_NOT_FOUND)

            // bcrypt 验证（D-03）
            val encoder = BCryptPasswordEncoder(12)
            if (!encoder.matches(password, user.passwordHash)) {
                throw UserException(BizCode.AUTH_FAILED)
            }

            val token = UUID.randomUUID().toString()
            return buildLoginResp(user.id!!, token)
        }

        /**
         * 构建 LoginResp。ChatService 通过 LoginResp.deviceType/deviceId 获取设备信息，
         * 无需重新解析 Request.params（Review 修复：避免重复反序列化）。
         *
         * deviceType/deviceId 从 LoginReq 的字段直接复制。
         */
        private fun buildLoginResp(userId: Long, token: String): LoginResp {
            return LoginResp.newBuilder()
                .setUserId(userId)
                .setUid(userId)
                .setToken(token)
                .setServerNow(System.currentTimeMillis())
                .build()
        }
    }
    ```

    2.2 **创建 RegisterHandler**：

    在 `gateway/src/main/kotlin/com/nebula/gateway/handler/user/` 下新建 `RegisterHandler.kt`：
    - 构造函数参数：`userRepository: UserRepository`, `idGenerator: SnowflakeIdGenerator`
    - 实现：检查用户名唯一性 → 检查密码长度(>=6) → bcrypt hash(12) → 创建 UserEntity 并 save → 生成 Token → 返回 RegisterResp

    ```kotlin
    class RegisterHandler(
        private val userRepository: UserRepository,
        private val idGenerator: SnowflakeIdGenerator
    ) : Handler<RegisterReq, RegisterResp> {

        override val method: String = "user/register"

        override suspend fun handle(req: RegisterReq): RegisterResp {
            val password = req.password
            if (password.length < 6) {
                throw UserException(BizCode.INVALID_PARAM, "密码长度不能少于 6 位")
            }

            val username = req.username.trim()
            if (userRepository.findByUsername(username) != null) {
                throw UserException(BizCode.USERNAME_EXISTS)
            }

            val encoder = BCryptPasswordEncoder(12)
            val passwordHash = encoder.encode(password)

            val user = UserEntity(
                username = username,
                passwordHash = passwordHash,
                nickname = req.nickname.ifBlank { username },
                avatar = req.avatar ?: ""
            )
            user.id = idGenerator.nextId()

            val saved = userRepository.save(user)
            return RegisterResp.newBuilder()
                .setUid(saved.id!!)
                .build()
        }
    }
    ```

    注意：RegisterResp 不需要返回 token（D-01: 注册后用户需登录获取 token）。RegisterResp 只有 uid 字段。

    3.1 **创建 SearchUserHandler** — method = "user/search"（BIZ-USER-01, D-07, D-08）：

    搜索范围：仅 username 字段，LIKE %keyword% 模糊匹配（D-07）。
    游标分页：每页最多 20 条，按注册时间（createdAt）倒序排列（D-08）。
    游标策略：首次 cursor=0，服务端取 limit+1 条判断 has_more。

    ```kotlin
    class SearchUserHandler(
        private val userRepository: UserRepository
    ) : Handler<SearchUserReq, SearchUserResp> {
        override val method: String = "user/search"

        override suspend fun handle(req: SearchUserReq): SearchUserResp {
            val keyword = req.keyword.trim()
            val cursor = req.cursor
            val limit = if (req.limit in 1..20) req.limit else 20

            val users = userRepository.findByUsernameContaining(
                keyword = keyword,
                cursor = cursor,
                limit = limit + 1  // 多取一条判断是否有更多
            )
            val hasMore = users.size > limit
            val result = if (hasMore) users.dropLast(1) else users

            val builder = SearchUserResp.newBuilder()
            result.forEach { entity ->
                builder.addUsers(UserBrief.newBuilder()
                    .setUid(entity.id!!)
                    .setUsername(entity.username)
                    .setDisplayName(entity.nickname)
                    .setAvatarUrl(entity.avatar)
                    .setCreatedAt(entity.createdAt?.toEpochMilli() ?: 0)
                    .build())
            }
            builder.setNextCursor(
                result.lastOrNull()?.createdAt?.toEpochMilli() ?: 0
            )
            builder.setHasMore(hasMore)
            return builder.build()
        }
    }
    ```

    3.2 **增强 UserRepository** — 添加模糊搜索 + 游标分页方法：

    在 `UserRepository.kt` 中添加：
    ```kotlin
    /**
     * 按用户名模糊搜索，支持游标分页（D-07, D-08）。
     *
     * 使用 JPA @Query 参数绑定（:keyword/:cursor）防止 SQL 注入（T-05-07 缓解措施）。
     * cursor=0 时忽略游标条件（首次查询）。
     *
     * @param keyword 搜索关键词（LIKE %keyword%）
     * @param cursor 游标（createdAt 毫秒时间戳），0 表示首次查询
     * @param limit 返回行数限制
     * @return 匹配的用户列表
     */
    @Query("SELECT u FROM UserEntity u WHERE u.username LIKE %:keyword% AND (:cursor = 0L OR u.createdAt &lt; :cursor) ORDER BY u.createdAt DESC")
    fun findByUsernameContaining(
        @Param("keyword") keyword: String,
        @Param("cursor") cursor: Long,
        limit: Int
    ): List<UserEntity>
    ```

    注意：
    - `&lt;` 是 `<` 的 XML 转义，在 Kotlin 中直接写 `<`
    - 实际 Kotlin 代码：`"SELECT u FROM UserEntity u WHERE u.username LIKE %:keyword% AND (:cursor = 0L OR u.createdAt < :cursor) ORDER BY u.createdAt DESC"`
    - cursor=0L 的特殊处理确保首次查询正确（Review 反馈#低：JPA 处理 0L 日期比较）
