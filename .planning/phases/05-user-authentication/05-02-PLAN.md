    ```kotlin
    /**
     * gRPC 双向流服务 — Nebula Chat 协议接入层（D-04, D-05）。
     *
     * 职责：
     * - 注册为 gRPC BindableService，方法名 "nebula.chat.ChatService/chat"
     * - 处理客户端 Envelope 消息（REQUEST/PING 方向）
     * - REQUEST: 调用 Dispatcher.dispatch() 分发
     * - PING: 返回 PONG 心跳应答
     * - 拦截 LoginResp：自动完成 Session 注册 + StreamObserver 绑定（D-05）
     * - 管理 StreamObserver 生命周期 + Session eviction callback 注册
     * - 处理 PUSH 方向的驱逐通知发送
     *
     * 设计决策引用：
     * - D-04: Gateway 层拦截 LoginResp，Handler 层不感知 StreamObserver
     * - D-05: LoginResp 拦截 → SessionRegistry.registerWithDeviceType() → 同类型踢下线 → LOGOUT 推送
     * - D-06: 无需 system/bind 请求
     *
     * @param dispatcher 请求分发器
     * @param sessionRegistry Session 注册中心
     * @param registry Handler 注册中心（用于查找方法是否存在）
     */
    class ChatService(
        private val dispatcher: Dispatcher,
        private val sessionRegistry: SessionRegistry,
        private val registry: HandlerRegistry
    ) : BindableService { ... }
    ```

    实现要点：

    a) **bindService()**: 使用 `ServerServiceDefinition.builder("nebula.chat.ChatService")`，添加一个 BIDI_STREAMING 方法 `chat`，使用 ProtoUtils.marshaller(Envelope.getDefaultInstance()) 作为编解码器。

    b) **chat()**: 返回 ObserverRequest 对。在 `ServerCalls.bidirectionalStreamingCall` 的 `FlowControlledRequestObserver` 或简单实现中：

    - 创建 `StreamObserver<Envelope>` 实例，在 `onNext()` 中处理：
      - `Direction.REQUEST` → `handleRequest(envelope)`
      - `Direction.PING` → 回复 PONG Envelope
      - 其他 → 忽略/日志

    - `handleRequest()` 方法：
      1. 调用 `dispatcher.dispatch(envelope.request)`
      2. 检查响应：若 `response.method == "user/login"` 且 `response.code == 200`，执行 D-05 拦截流程：
         - 反序列化 LoginResp: `LoginResp.parseFrom(response.result.toByteArray())`
         - 构建 Session: 从 LoginReq（缓存在 Request.params 中已反序列化或重新 parse）获取 deviceType。注意：由于 LoginHandler 是纯业务 Handler，ChatService 需要从 Request.params 重新解析 LoginReq 以获取 deviceType 和 deviceId。
         - 或者更简单：LoginReq 的 deviceType 信息在 ChatService 接收到 Envelope 时已存为成员。在 `handleRequest()` 中，从 Request.params 解析 LoginReq 获取 deviceType。
         - 调用 `sessionRegistry.registerWithDeviceType(session, currentObserver)`
         - 获取驱逐回调结果：若旧 token 被驱逐，通过 evictionCallbacks 发送 LOGOUT 通知（PUSH Envelope 含 PushEventType.LOGOUT）到旧 StreamObserver 并关闭
         - 发送 LoginResp Envelope（Direction.RESPONSE）给新的 currentObserver
      3. 若非 login 方法，直接发送 Response Envelope

    c) **eviction callback 注册**：在 ChatService 构造函数或第一个请求到达时，在 SessionRegistry 上注册 eviction callback：
      ```kotlin
      sessionRegistry.onEviction { token ->
          // 通过 token 查找对应的 StreamObserver（需要维护 token→observer 映射）
          // 发送 PUSH LOGOUT Envelope → close stream
          val observer = observerMap.remove(token)
          if (observer != null) {
              observer.onNext(logoutEnvelope)
              observer.onCompleted()
          }
      }
      ```
      需要维护 `tokenToObserver: ConcurrentHashMap<String, StreamObserver<Envelope>>` 映射。

    d) **connectionId 生成**：使用 UUID.randomUUID().toString()。

    e) **gRPC 远程地址获取**：ChatService 可以通过 `Grpc.TRANSPORT_ATTR_REMOTE_ADDR` 获取客户端 IP（用于 IP 限流）。在创建请求处理时提取并存入 metadata 或传递给 RateLimitInterceptor。

    f) **PING 处理**：收到 Direction.PING 时，回复 Envelope(direction=PONG)。

    2.1 **创建 LoginHandler** — `gateway/src/main/kotlin/com/nebula/gateway/handler/user/LoginHandler.kt`（需要新建 user 子目录）：

    ```kotlin
    /**
     * 登录 Handler — method = "user/login"（AUTH-01, AUTH-02, D-04, D-05）。
     *
     * 职责边界：
     * - 仅处理密码验证（bcrypt cost 12）或 Token 重连验证（AUTH-02）
     * - 返回 LoginResp（含 token、uid、server_now、last_read_info、server_last_msg_id）
     * - 不感知 StreamObserver，不处理 Session 注册（D-04）
     *
     * Session 注册和 StreamObserver 绑定由 ChatService 在检测到 LoginResp 后完成。
     *
     * @param userRepository 用户数据仓库
     * @param sessionRegistry Session 注册中心（用于 Token 重连验证）
     * @param idGenerator Snowflake ID 生成器（初始化用，保留备用）
     */
    class LoginHandler(
        private val userRepository: UserRepository,
        private val sessionRegistry: SessionRegistry,
        private val idGenerator: SnowflakeIdGenerator
    ) : Handler<LoginReq, LoginResp> {

        override val method: String = "user/login"

        override suspend fun handle(req: LoginReq): LoginResp {
            // 场景 1: Token 重连（AUTH-02）
            if (req.hasToken()) {
                val token = req.token
                val existingSession = sessionRegistry.validate(token)
                if (existingSession != null) {
                    // Token 有效，返回已有 Session 的 LoginResp
                    val user = userRepository.findById(existingSession.userId)
                        ?: throw UserException(BizCode.USER_NOT_FOUND)
                    return buildLoginResp(existingSession.userId, token)
                }
            }

            // 场景 2: 用户名+密码登录
            val username = req.username ?: throw UserException(BizCode.INVALID_PARAM)
            val password = req.password ?: throw UserException(BizCode.INVALID_PARAM)
            val user = userRepository.findByUsername(username)
                ?: throw UserException(BizCode.USER_NOT_FOUND)

            // bcrypt 验证（D-03）
            val encoder = BCryptPasswordEncoder(12)
            if (!encoder.matches(password, user.passwordHash)) {
                throw UserException(BizCode.AUTH_FAILED)
            }

            val token = UUID.randomUUID().toString()
            return buildLoginResp(user.id!!, token)
        }

        private fun buildLoginResp(userId: Long, token: String): LoginResp {
            return LoginResp.newBuilder()
                .setUserId(userId)
                .setUid(userId)
                .setToken(token)
                .setServerNow(System.currentTimeMillis())
                // last_read_info 和 server_last_msg_id 在 Phase 7 填充，当前为空
                .build()
        }
    }
    ```

    ```kotlin
    /**
     * 注册 Handler — method = "user/register"（D-01, D-02, D-03）。
     *
     * 注册流程：
     * 1. 校验用户名唯一性（UserRepository.findByUsername）
     * 2. 校验密码强度（最少 6 位）
     * 3. bcrypt 哈希密码（cost 12, D-03）
     * 4. 创建 UserEntity（Snowflake 生成 ID）
     * 5. 持久化到 MySQL
     *
     * IP 频率限制（每小时 5 次）由 RateLimitInterceptor 处理。
     *
     * @param userRepository 用户数据仓库
     * @param idGenerator Snowflake ID 生成器
     */
    class RegisterHandler(
        private val userRepository: UserRepository,
        private val idGenerator: SnowflakeIdGenerator
    ) : Handler<RegisterReq, RegisterResp> {

        override val method: String = "user/register"

        override suspend fun handle(req: RegisterReq): RegisterResp {
            // 密码强度校验（D-02: 最低 6 位）
            val password = req.password
            if (password.length < 6) {
                throw UserException(BizCode.INVALID_PARAM, "密码长度不能少于 6 位")
            }

            // 用户名唯一性校验（D-02）
            val username = req.username.trim()
            if (userRepository.findByUsername(username) != null) {
                throw UserException(BizCode.USERNAME_EXISTS)
            }

            // bcrypt 哈希（D-03）
            val encoder = BCryptPasswordEncoder(12)
            val passwordHash = encoder.encode(password)

            // 创建用户实体
            val user = UserEntity(
                username = username,
                passwordHash = passwordHash,
                nickname = req.nickname.ifBlank { username },
                avatar = req.avatar ?: ""
            )
            user.id = idGenerator.nextId()

            val saved = userRepository.save(user)

            val token = UUID.randomUUID().toString()
            return RegisterResp.newBuilder()
                .setUid(saved.id!!)
                .setToken(token)
                .build()
        }
    }
    ```

    3.1 **创建 SearchUserHandler** — `gateway/src/main/kotlin/com/nebula/gateway/handler/user/SearchUserHandler.kt`：

    ```kotlin
    /**
     * 搜索用户 Handler — method = "user/search"（BIZ-USER-01, D-07, D-08）。
     *
     * 搜索范围：
     * - 仅 username 字段，LIKE %keyword% 模糊匹配（D-07）
     * - 游标分页，每页最多 20 条，按注册时间（createdAt）倒序排列（D-08）
     *
     * 游标策略：
     * - 首次查询 cursor=0，服务端按 createdAt DESC 取 limit+1 条
     * - 返回 next_cursor = 最后一页最后一条的 createdAt
     * - 客户端翻页时传入上一页返回的 next_cursor
     * - has_more=true 表示还有更多数据
     *
     * @param userRepository 用户数据仓库
     */
    class SearchUserHandler(
        private val userRepository: UserRepository
    ) : Handler<SearchUserReq, SearchUserResp> {

        override val method: String = "user/search"

        override suspend fun handle(req: SearchUserReq): SearchUserResp {
            val keyword = req.keyword.trim()
            val cursor = req.cursor  // 0 表示首次查询
            val limit = if (req.limit in 1..20) req.limit else 20

            val users = userRepository.findByUsernameContaining(
                keyword = keyword,
                cursor = cursor,
                limit = limit + 1  // 多取一条判断是否有更多
            )

            val hasMore = users.size > limit
            val result = if (hasMore) users.dropLast(1) else users

            val builder = SearchUserResp.newBuilder()
            result.forEach { entity ->
                builder.addUsers(UserBrief.newBuilder()
                    .setUid(entity.id!!)
                    .setUsername(entity.username)
                    .setDisplayName(entity.nickname)
                    .setAvatarUrl(entity.avatar)
                    .build())
            }
            builder.setNextCursor(
                result.lastOrNull()?.createdAt?.toEpochMilli() ?: 0
            )
            builder.setHasMore(hasMore)
            return builder.build()
        }
    }
    ```
