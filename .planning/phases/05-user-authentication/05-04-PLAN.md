    1.3 **重写 registerHandlers() 函数 —— Review 修复：消除参数列表膨胀：**

    原方案：为每个 Handler 声明显式参数（10 个参数），维护负担高。
    新方案：**从 Koin 容器中迭代所有 Handler，自动注册。**

    ```kotlin
    /**
     * 将 Koin 中注册的所有 Handler 自动注册到 HandlerRegistry。
     *
     * 使用 Koin 的 getAll() 获取所有 Handler<*, *> 实例，遍历注册。
     * Review 修复：替代原每个 Handler 显式参数的方案，消除参数列表膨胀。
     * 新增 Handler 只需在 handlerModule 中添加 single { } 声明即可自动注册。
     *
     * @param registry HandlerRegistry 实例
     * @param protoCodec ProtoCodec 实例
     * @param koin Koin 容器实例
     */
    fun registerHandlers(
        registry: HandlerRegistry,
        protoCodec: ProtoCodec,
        koin: Koin
    ) {
        // 通过 Koin.getAll() 获取所有 Handler 实例（Review 修复：迭代自动注册）
        val handlers = koin.getAll<Handler<*, *>>()
        handlers.forEach { handler ->
            // 通过 ProtoCodec.buildCodec 获取序列化方法引用
            // 需要从 Handler 的泛型参数中获取 Req/Resp 的 KClass
            // 方案：在 HandlerEntry 中保存 reqClass/respClass，由 ProtoCodec 在注册时解析
            val reqClass = handler.reqClass()  // 需要 Handler 提供 reqClass/respClass 方法
            val respClass = handler.respClass()
            val reqCodec = ProtoCodec.buildCodec(reqClass)
            val respCodec = ProtoCodec.buildCodec(respClass)
            registry.register(HandlerEntry(
                handler = handler,
                reqClass = reqClass,
                respClass = respClass,
                parseFrom = reqCodec.parseFrom,
                toByteArray = respCodec.toByteArray
            ))
        }
    }
    ```

    **注意：** 上述方案需要 Handler 接口提供 `reqClass()` 和 `respClass()` 方法。如果 Handler 接口尚未定义这些方法：

    **备选方案 A** — 在 Handler 接口中添加抽象方法：
    ```kotlin
    interface Handler<Req : Any, Resp : Any> {
        val method: String
        val reqClass: KClass<Req>
        val respClass: KClass<Resp>
        suspend fun handle(req: Req): Resp
    }
    ```
    每个 Handler 实现类需要添加：
    ```kotlin

    override val respClass: KClass<LoginResp> = LoginResp::class
    ```

    此方案改动 Handler 接口，影响 Phase 4。

    **备选方案 B（推荐）** — 使用 `HandlerRegistry.register()` 的重载，存储一个 `handler: Handler<*, *>` 并利用 KClass 反射（或直接存储 Class 引用）：

    **最简单的实现（推荐）：**
    在 `GatewayModule.kt` 中添加一个辅助函数，为每个 Handler 手动构建 HandlerEntry。虽然仍是显式注册，但通过一个 `registerHandler()` 辅助函数显著减少模板代码：

    ```kotlin
    /**
     * 注册单个 Handler 到 HandlerRegistry 的模板辅助方法。
     *
     * @param registry HandlerRegistry 实例
     * @param handler Handler 实例
     * @param reqClass 请求类型的 KClass
     * @param respClass 响应类型的 KClass
     */
    private inline fun <reified ReqT : Any, reified RespT : Any> registerHandler(
        registry: HandlerRegistry,
        handler: Handler<ReqT, RespT>
    ) {
        val reqCodec = ProtoCodec.buildCodec(ReqT::class)
        val respCodec = ProtoCodec.buildCodec(RespT::class)
        registry.register(
            HandlerEntry(
                handler = handler,
                reqClass = ReqT::class,
                respClass = RespT::class,
                parseFrom = reqCodec.parseFrom,
                toByteArray = respCodec.toByteArray
            )
        )
    }
    ```

    然后在 registerHandlers() 中使用：
    ```kotlin
    fun registerHandlers(registry: HandlerRegistry, protoCodec: ProtoCodec, pingHandler: PingHandler, loginHandler: LoginHandler, ...) {
        registerHandler(registry, pingHandler)
        registerHandler(registry, loginHandler)
        // ...
    }
    ```
