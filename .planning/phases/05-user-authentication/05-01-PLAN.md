    ```kotlin
    /**
     * 按设备类型注册 Session — 同类型设备互踢（D-05, AUTH-05）。
     *
     * 流程：
     * 1. 检查 deviceTypeIndex 是否存在同 userId+deviceType 的旧 token
     * 2. 若存在，unregister 旧 token（触发驱逐回调 → LOGOUT 推送）
     * 3. 注册新 Session（L1 + L2 token key）
     * 4. 写入设备类型交叉引用到 Redis（key: "session:{userId}:{deviceType}" → token, TTL 7天）
     * 5. 更新本地 deviceTypeIndex
     *
     * @param session 新 Session
     * @return 被驱逐的旧 Session token，若无则 null
     */
    suspend fun registerWithDeviceType(session: Session): String? {
        val key = deviceTypeKey(session.userId, session.deviceType)
        val existingToken = deviceTypeIndex[key]

        if (existingToken != null) {
            // 同设备类型的旧连接存在，触发踢下线（LOGOUT 推送在 eviction callback 中完成）
            unregister(existingToken)
        }

        // 注册新 Session（L1 + L2 token key）
        register(session)

        // 写入设备类型交叉引用到 Redis
        saveDeviceTypeMapping(session)

        // 更新本地索引
        deviceTypeIndex[key] = session.token
        return existingToken
    }
    ```

    3.3 **RateLimitInterceptor 增强 — 注册 IP 限流（含内存泄漏修复）**：

    a) 在 `RateLimitInterceptor.kt` 中新增 `private val registerLimiter = RegisterRateLimiter()` 字段
    b) 在 `intercept()` 方法中增加判断：如果 `request.method == "user/register"`，使用 RegisterRateLimiter 进行 IP 限流检查（D-02）。IP 来源：`request.metadataMap["x-client-ip"] ?: request.metadataMap["x-forwarded-for"] ?: "unknown"`
    c) 新增内部类 `RegisterRateLimiter`：

    ```kotlin
    /**
     * 注册 IP 限流器 — 每小时每 IP 最多 5 次注册（D-02）。
     *
     * 注意：内存泄漏防护 — tryAcquire() 每次调用后检查 ipRequestTimes 中空列表的条目并移除（Review 修复）。
     */
    class RegisterRateLimiter {
        private val ipRequestTimes = ConcurrentHashMap<String, MutableList<Long>>()
        private val maxRequests = 5
        private val windowMs = 60 * 60 * 1000L  // 1 hour

        fun tryAcquire(ip: String): Boolean {
            val now = System.currentTimeMillis()
            val times = ipRequestTimes.getOrPut(ip) { mutableListOf() }
            synchronized(times) {
                times.removeAll { now - it > windowMs }
                if (times.size >= maxRequests) return false
                times.add(now)
            }
            // Review 修复：如果 times 列表为空，清理 IP 条目防止内存泄漏
            // 避免恶意 IP 变换导致 ipRequestTimes 无限膨胀
            if (times.isEmpty()) {
                ipRequestTimes.remove(ip)
            }
            return true
        }
    }
    ```
    注意：`synchronized(times)` 块结束后、`if (times.isEmpty())` 之前在 `synchronized` 外部执行 remove — 因为 `synchronized` 内刚 add(now)，times 不可能为空。实际清理逻辑应调整为：在 tryAcquire() 返回 false 后（次数超限），同时在每次调用时惰性清理已过期条目。

    修正后的正确逻辑：
    ```kotlin
    fun tryAcquire(ip: String): Boolean {
        val now = System.currentTimeMillis()
        val times = ipRequestTimes.getOrPut(ip) { mutableListOf() }
        synchronized(times) {
            times.removeAll { now - it > windowMs }
            if (times.size >= maxRequests) return false
            times.add(now)
            // 清理空列表条目（内存泄漏修复）
            if (times.isEmpty()) {
                // 惰性清理：当列表为空时移除此 IP 条目
            }
        }
        // 在 synchronized 外部检查是否要清理
        if (times.isEmpty()) {
            ipRequestTimes.remove(ip, times)
        }
        return true
    }
    ```
    简单实现：在 `synchronized` 块末尾加上一个额外判断，若 `times.removeAll` 后列表为空且 `maxRequests` 已达上限，则一定移除条目。但更可靠的惰性清理方法是：每次调用时，在 synchronized 内的 `removeAll` 之后检查，若列表为空则 remove IP。但此时调用方已退出（add 在检查之前）。

    最终实现：在同步块结尾检查 `if (times.isEmpty()) ipRequestTimes.remove(ip, times)` — 注：只有 `times.removeAll` 移除了全部条目而 `maxRequests > 0` 且未 add 时才会触发，即当 `ip` 条目上次窗口已全部过期、但本次不再有请求的情况下。实际上 getOrPut 会创建一个空列表，在 add 之前检查并清除。

    最简洁且正确的实现（Review 建议的修复方案）：
    ```kotlin
    fun tryAcquire(ip: String): Boolean {
        val now = System.currentTimeMillis()
        val times = ipRequestTimes.getOrPut(ip) { mutableListOf() }
        synchronized(times) {
            times.removeAll { now - it > windowMs }
            if (times.size >= maxRequests) return false
            times.add(now)
        }
        // 内存泄漏修复：清除已过期空的 IP 条目（Review 反馈#5）
        if (times.isEmpty()) {
            ipRequestTimes.remove(ip, times)
        }
        return true
    }
    ```
