    ```kotlin
    /**
     * 从请求的 metadata map 中提取 Token。
     *
     * Token 通过 Request.metadata 传递（D-04 推荐），key 为 "authorization"。
     * 客户端应在每个认证请求的 Envelope.Request.metadata 中设置 "authorization" → token。

     *
     * @param request 客户端请求

    override fun extractToken(request: Request): String? {
        val metadataMap = request.metadataMap
        val token = metadataMap["authorization"]
        if (token.isNullOrBlank()) return null
        return token
    }
    ```

    - 新增内部类 `RegisterRateLimiter`（IP → 请求时间戳队列，每小时最多 5 次，60 分钟窗口）：
      ```kotlin
      /**
       * 注册 IP 限流器 — 每小时每 IP 最多 5 次注册（D-02）。
       */
      class RegisterRateLimiter {
          private val ipRequestTimes = ConcurrentHashMap<String, MutableList<Long>>()
          private val maxRequests = 5
          private val windowMs = 60 * 60 * 1000L

          fun tryAcquire(ip: String): Boolean {
              val now = System.currentTimeMillis()
              val times = ipRequestTimes.getOrPut(ip) { mutableListOf() }
              synchronized(times) {
                  times.removeAll { now - it > windowMs }
                  if (times.size >= maxRequests) return false
                  times.add(now)
                  return true
              }
          }
      }
      ```
