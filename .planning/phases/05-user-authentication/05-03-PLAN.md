    ```kotlin
    /**
     * 获取用户详细资料 Handler — method = "user/getProfile"（BIZ-USER-02）。
     *
     * 根据目标用户 UID 返回详细资料页面所需的所有公开信息。
     * 当前可查询任意用户，Phase 8 可能添加好友/非好友的不同可见范围。
     *
     * @param userRepository 用户数据仓库
     */
    class GetProfileHandler(
        private val userRepository: UserRepository
    ) : Handler<GetProfileReq, GetProfileResp> {

        override val method: String = "user/getProfile"

        override suspend fun handle(req: GetProfileReq): GetProfileResp {
            val user = userRepository.findById(req.uid)
                ?: throw UserException(BizCode.USER_NOT_FOUND)

            return GetProfileResp.newBuilder()
                .setUid(user.id!!)
                .setUsername(user.username)
                .setDisplayName(user.nickname)
                .setAvatarUrl(user.avatar)
                .setCreatedAt(user.createdAt?.toEpochMilli() ?: 0)
                .build()
        }
    }
    ```

    注意：UserEntity 当前没有 gender/bio 字段。GetProfileResp 中的 gender(5) 和 bio(6) 在 v1 中暂不填充，后续扩展。当前仅返回已有的字段。

    1.2 **创建 BatchGetUserHandler** — method = "user/batchGet"（BIZ-USER-03）：

    ```kotlin
    /**
     * 批量获取用户摘要 Handler — method = "user/batchGet"（BIZ-USER-03）。
     *
     * 根据 UID 列表批量返回用户简要信息。
     * 用于会话成员列表、好友列表等需要展示多个用户摘要的场景。
     *
     * @param userRepository 用户数据仓库
     */
    class BatchGetUserHandler(
        private val userRepository: UserRepository
    ) : Handler<BatchIdRequest, BatchGetUserResp> {

        override val method: String = "user/batchGet"

        override suspend fun handle(req: BatchIdRequest): BatchGetUserResp {
            val users = userRepository.findAllById(req.uidsList.map { it })
            val builder = BatchGetUserResp.newBuilder()

            users.forEach { entity ->
                builder.addUsers(UserBrief.newBuilder()
                    .setUid(entity.id!!)
                    .setUsername(entity.username)
                    .setDisplayName(entity.nickname)
                    .setAvatarUrl(entity.avatar)
                    .build())
            }
            return builder.build()
        }
    }
    ```

    2.1 **创建 BatchGetStatusHandler** — method = "user/batchGetStatus"（BIZ-USER-04, D-10）：

    ```kotlin
    /**
     * 批量查询用户在线状态 Handler — method = "user/batchGetStatus"（BIZ-USER-04, D-10）。
     *
     * 查询逻辑：
     * - 遍历请求中的 UID 列表
     * - 跳过 hide_online_status=true 的用户（D-10 隐私过滤）
     * - 对可见用户，查询 OnlineStatusRepository.isOnline()
     * - 返回在线状态列表（status: 0=离线, 1=在线）
     *
     * @param onlineStatusRepository 在线状态缓存
     * @param privacyRepository 隐私设置缓存
     */
    class BatchGetStatusHandler(
        private val onlineStatusRepository: OnlineStatusRepository,
        private val privacyRepository: PrivacyRepository
    ) : Handler<BatchIdRequest, BatchGetStatusResp> {

        override val method: String = "user/batchGetStatus"

        override suspend fun handle(req: BatchIdRequest): BatchGetStatusResp {
            val builder = BatchGetStatusResp.newBuilder()

            for (uid in req.uidsList) {
                // D-10: 跳过 hide_online_status=true 的用户
                if (privacyRepository.getHideOnlineStatus(uid)) continue

                val isOnline = onlineStatusRepository.isOnline(uid)
                builder.addStatuses(UserOnlineStatus.newBuilder()
                    .setUid(uid)
                    .setStatus(if (isOnline) 1 else 0)
                    .build())
            }
            return builder.build()
        }
    }
    ```

    注意：OnlineStatusRepository 的 isOnline() 方法目前是简单的 key 存在检测。
    在 Phase 8 会增加 last_active_at 等字段和伪在线逻辑。

    2.2 **创建 SetPrivacyHandler** — method = "user/setPrivacy"（BIZ-USER-05, D-09, D-11）：

    ```kotlin
    /**
     * 设置隐私选项 Handler — method = "user/setPrivacy"（BIZ-USER-05, D-09, D-11）。
     *
     * 写入策略（D-09）：
     * - 写 Redis 立即生效（本用户和其他用户查询时立即生效）
     * - 异步刷 MySQL 做持久化（Pitfall 4: 接受宕机时最近的写入丢失，Redis 为主）
     *
     * @param privacyRepository 隐私设置缓存
     */
    class SetPrivacyHandler(
        private val privacyRepository: PrivacyRepository
    ) : Handler<SetPrivacyReq, Response> {

        override val method: String = "user/setPrivacy"

        override suspend fun handle(req: SetPrivacyReq): Response {
            val session = coroutineContext.requireSession()
            privacyRepository.setHideOnlineStatus(session.userId, req.hideOnlineStatus)
            return Response.newBuilder()
                .setCode(BizCode.OK.code)
                .setMsg("ok")
                .setMethod(method)
                .build()
        }
    }
    ```

    注意：SetPrivacyHandler 返回 Response（通用成功响应）而非特定 proto 消息。因为 ProtoCodec 需要 Resp 类型匹配 HandlerEntry，所以返回 `Response` (com.nebula.chat.Response) 作为通用成功响应。

    ```kotlin
    /**
     * 获取隐私选项 Handler — method = "user/getPrivacy"（BIZ-USER-06, D-11）。
     *
     * 读取策略（D-11）：
     * - 优先读 Redis（即时生效）
     * - Redis 未命中则从 MySQL 读取并写回 Redis（Pitfall 4 恢复）
     *
     * 请求无需入参，userId 从 Session 获取。
     *
     * @param privacyRepository 隐私设置缓存
     */
    class GetPrivacyHandler(
        private val privacyRepository: PrivacyRepository
    ) : Handler<GetPrivacyReq, GetPrivacyResp> {

        override val method: String = "user/getPrivacy"

        override suspend fun handle(req: GetPrivacyReq): GetPrivacyResp {
            val session = coroutineContext.requireSession()
            val hideOnlineStatus = privacyRepository.getHideOnlineStatus(session.userId)
            return GetPrivacyResp.newBuilder()
                .setHideOnlineStatus(hideOnlineStatus)
                .build()
        }
    }
    ```
